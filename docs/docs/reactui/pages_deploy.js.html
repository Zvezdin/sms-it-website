import React from "react"

import { notify, cycleProcessingMessages } from "../helper_functions"

import StageBar from "../components/stage-bar"
import PhoneButton from "../components/phone-button"
import ErrorAlert from "../components/error-alert"
import HelpButton from "../components/help-button"

import arrow from "../images/left-chevron.png"
import "../styles/default.css"

import store from "../Store"


/**
 * Responsible for rendering the deploy page on screen, where the user can select a phone number for
 * an undeployed bot. Consists of 'Deploy <bot name>' heading, stage bar with 'Phone Number' text in
 * green, and a list of phone numbers implemented with the phone-button component.
 * 
 * @returns the deploy page for undeployed bots.
 */
class Deploy extends React.Component {
    helpURL = "https://sms-it.io/project/for-users/07-deploy/"

    processingMessages = [
        "PROCESSING", "PROCESSING.", "PROCESSING..", "PROCESSING...",
        "PLEASE WAIT", "PLEASE WAIT.", "PLEASE WAIT..", "PLEASE WAIT...",
        "DEPLOYING TO NUMBER", "DEPLOYING TO NUMBER.", "DEPLOYING TO NUMBER..", "DEPLOYING TO NUMBER...",
    ]

    constructor(props) {
        super(props);
        this.state = {
            processing: false,
            msgIdx: 0,
            showError: false
        }
    }


    /**
     * Invoked immediately after the current page is mounted (inserted into the DOM).
     */
    componentDidMount() {
        cycleProcessingMessages(this)
    }


    /**
     * Reference to a function that deploys the selected bot onto a given phone number.
     * This is passed to the phone-button component and is called when the user selects an available number
     * to deploy the bot to.
     * 
     * @param {number} number - the phone number the bot is to be assigned to.
     */
    deployAsNew = async (number) => {
        this.setState({ processing: true })

        store.candidateBot.phoneNumber = number

        await store.linkCandidateBotAsNew()
        notify(
            "success",
            "Success!",
            (store.candidateBot.name + " successfully deployed on " + store.candidateBot.phoneNumber),
            5000
        )

        this.setState({ processing: false })
        this.props.history.push("/manage")
    }


    /**
     * Reference to a function that sets the processing flag to false, and the showError flag to true.
     * This is passed to the phone-button component and is called when an error occurs - it allows
     * the child phone-button component to change the state of this parent page in order to show the error
     * popup component on screen.
     */
    showError = () => {
        this.setState({ processing: false, showError: true })
    }

    
    render() {
        let processingText = this.processingMessages[this.state.msgIdx % this.processingMessages.length]

        return (
            <>
                {
                    this.state.showError 
                    ? <ErrorAlert 
                        name="Error deploying chatbot" 
                        message="Something went wrong with Twilio while trying to deploy the chatbot" 
                        _callback={() => this.setState({ showError: false, processing: false })} /> 
                    : null
                }
                
                <div className="column center-items" style={{height: "15%"}}>
                    <h2>Deploy {store.candidateBot.name}</h2>
                </div>

                <div 
                    data-testid="buttonList" 
                    className="container column" 
                    style={
                        this.state.processing
                        ? {display: "none"}
                        : {display: "flex", height: "auto", justifyContent: "flex-start"}
                    }
                >

                    <StageBar current={1} />

                    <h3 style={{margin: "20px"}}>Select a number to deploy to</h3>

                    {store.phoneNumbers.map((number, index) => 
                        <PhoneButton 
                            key={number} 
                            number={number} 
                            handleAvailable={this.deployAsNew}
                            showError={this.showError} />
                    )}

                    {
                        store.phoneNumbers.length < 1
                        ? (
                            <p style={{fontSize: "larger", color: "firebrick"}}>
                                There are no phone numbers on this Twilio account!
                            </p>
                        ) : null
                    }

                </div>

                <div 
                    className="container" 
                    style={
                        this.state.processing 
                        ? {display: "flex", height: "85%"} 
                        : {display: "none"}
                    }
                >
                    <div 
                        className={this.state.processing ? "big-loader" : ""} 
                        style={
                            this.state.processing 
                            ? {display: "flex", margin: "60px"} 
                            : {display: "none", margin: "60px"}
                        } 
                    />
                    <div className={this.state.processing ? "processing" : "hidden"}>
                        {processingText}
                    </div>
                </div>

                <div className="icon-container" style={{left: 0}}>
                    <img draggable="false" src={arrow} alt="Back" className="arrow-icon" onClick={() => {
                        this.props.history.push({pathname: "/test", state: {deploy: true}})
                    }} />
                </div>

                <HelpButton url={this.helpURL} />
            </>
        )
    }

}

export default Deploy;
