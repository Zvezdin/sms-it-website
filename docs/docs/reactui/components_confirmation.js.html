import React from "react"

import { cycleProcessingMessages } from '../helper_functions'

import { INTENTION } from "../constants/manage"

import ErrorAlert from "./error-alert"

import attention from "../images/attention.png"
import "../styles/default.css"


/**
 * Displays a confirmation prompt over the current page.
 * The page is darkened with a transparent overlay to put focus on the prompt.
 * 
 * @param {INTENTION} intent - the action that needs confirmation (enum value: 1 UNLINK/2 DELETE).
 * @param {string} message - the message to display on the prompt.
 * @param {function} accept - reference to the function to call when 'Yes' is clicked.
 * @param {function} cancel - reference to the function to call when 'Cancel' is clicked.
 * 
 * @returns confirmation prompt to user.
 */
class Confirmation extends React.Component {
    processingMessages = [
        "PROCESSING", "PROCESSING.", "PROCESSING..", "PROCESSING...",
        "PLEASE WAIT", "PLEASE WAIT.", "PLEASE WAIT..", "PLEASE WAIT...",
    ]

    constructor(props) {
        super(props)
        this.state = {
			processing: false,
            msgIdx: 0,
            showError: false
        }
    }


    /**
     * Invoked immediately after the current component is mounted (inserted into the DOM).
     */
    componentDidMount() {
        cycleProcessingMessages(this)
    }


    /**
     * Produces the top label on the confirmation prompt using the intent argument.
     * 
     * @returns a banner that describes the action that needs confirmation.
     */
    getBanner() {
        let heading;
        if (this.props.intent === INTENTION.UNLINK) { heading = "UNLINKING CHATBOT" }
        else if (this.props.intent === INTENTION.DELETE) { heading = "DELETING CHATBOT" }

        return (
            <div data-testid="heading" className="confirm-banner">
                <img draggable="false" src={attention} alt="Warning!" height="28px" style={{margin: "10px 12px 10px 0px"}} />
                {heading}
            </div>
        )
    }


    render() {
        let processingText = this.processingMessages[this.state.msgIdx % this.processingMessages.length]

        return (
            <div className="dark-overlay container">
                {
                    this.state.showError 
                    ? <ErrorAlert 
                        name="Unable to delete chatbot" 
                        message="Something messed up within Twilio"  
                        _callback={() => this.setState({ showError: false })} />
                    : null
                }

                <div 
                    className="confirm container column" 
                    style={{justifyContent: "space-between"}} 
                >

                    {this.getBanner()}

                    <div className="container column">

                        <div 
                            data-testid="content" 
                            className="container column confirm-content" 
                            style={
                                this.state.processing 
                                ? {display: "none"} 
                                : {display: "flex"}
                            }
                        >

                            <span className="confirm-text" style={{fontWeight: "500", fontSize: "larger"}}>Are you sure?</span>

                            {this.props.message}

                            <div className="container row" style={{margin: 0, height: "auto", width: "400px", marginTop: "12px"}}>
                                <button 
                                    className="binary-option" 
                                    style={{color: "#B22222", backgroundColor: "#F3CECE"}} 
                                    onClick={() => this.props.cancel()}
                                >CANCEL</button>

                                <button 
                                    className="binary-option" 
                                    style={{color: "#008000", backgroundColor: "#CEF2CE"}} 
                                    onClick={async () => {
                                        this.setState({ processing: true })

                                        await this.props.accept()

                                        this.setState({ processing: false })
                                }}>YES</button>
                            </div>

                        </div>

                        <div 
                            className={this.state.processing ? "processing" : "hidden"}
                            style={{position: "relative", margin: "5px"}}
                        >
                            {processingText}
                        </div>
                        <div 
                            data-testid="spinner" 
                            className={this.state.processing ? "medium-loader" : ""} 
                            style={this.state.processing ? {display: "flex", marginBottom: "20px"} : {display: "none"}} 
                        />

                    </div>
                    
                </div>

            </div>
        )
    }

}

export default Confirmation;
