import React from "react"

import "../styles/default.css"


/**
 * Displays a stage bar with stages separated by chevrons to indicate the order of stages. The current stage has letters 
 * outlined in green, while other stages are greyed out.
 * 
 * @param {number} current - the index of the current stage in the array defined within this component: ["Test as SMS Chatbot", "Phone Number"].
 * 
 * @returns a stage bar with the current stage text outlined in green.
 */
class StageBar extends React.Component {

    constructor(props) {
        super(props);
        this.stages = ["Test as SMS Chatbot", "Phone Number"];
    }

    
    render() {
        return (
            <div className="container row" style={{height: "auto", margin: 0, width: "50%", maxWidth: "800px", transform: "translateY(-15px)"}}>
                {this.stages.map((stage, index) => (
                    [
                        <li key={"stage"+index} className={index === this.props.current ? "stage-coloured" : "stage-greyed"}>{stage}</li>,
                        index < this.stages.length-1 ? <div key={"arrow"+index} className="right-chevron" /> : null
                    ]
                ))}
            </div>
        )
    }

}

export default StageBar;
