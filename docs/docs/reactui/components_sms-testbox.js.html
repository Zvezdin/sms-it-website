import React from "react"

import ErrorAlert from "../components/error-alert"
import { log } from "../nodeImports"

import store from "../Store"


/**
 * Displays a smartphone with a text area and send button for input. Allows the user to interact and have a coversation
 * with a simulated version of the selected bot. User input appears on the right side of the smartphone, with bot replies
 * appearing on the left. Used for chatbot testing purposes.
 * 
 * @returns a smartphone with a user input field that displays conversation history with a specific bot.
 */
class SMSTestBox extends React.Component {

    constructor(props) {
        super(props)
        this.state = {
            msg: "",
            messages: [],
            showError: false
        }
    }


    /**
     * Takes the current user input, clears it from the input textarea, and sends it to a simulated version of the bot.
     * The bot reply is concatenated to the component state 'messages' array and is displayed in the smartphone SMS conversation.
     */
    async sendMessageAndGetReply() {
        if (this.state.msg === "") { return }

        this.state.messages.push({ msg: this.state.msg, isBot: false })
        await this.setState(this.state)

        this.inputField.value = ""
        
        let responses
        try {
            responses = await store.simulateCandidateBot(this.state.msg)
        } catch (e) {
            log.error(e)
            this.setState({ showError: true })
            return
        }
        this.state.messages = this.state.messages.concat(responses.map(m => {return {msg: m, isBot: true}}))
        await this.setState(this.state)

        this.scrollToBottom()
    }


    /**
     * Filter newline characters from user input. This is needed because users can send messages using the 'Enter' key.
     */
    handleInputChange = e => {
        const filteredMsg = e.target.value.replace(['\n'], '')
        this.setState({ msg: filteredMsg })
    }


    /**
     * Scrolls the chat window to the bottom/most recent message.
     */
    scrollToBottom() {
        this.messagesEnd.scrollIntoView({ behavior: "smooth" })
    }


    render() {
        return (
            <div data-testid="testbox" className="sms-box" style={{height: "564px", width: "284px", padding: "30px 24px 24px 24px"}}>
                {
                    this.state.showError 
                    ? <ErrorAlert 
                        name="Error sending message to chatbot" 
                        message="Something messed up within Twilio" 
                        _callback={() => this.setState({ showError: false })} /> 
                    : null
                }

                <div data-testid="chatbox" className="container column" style={{justifyContent: "flex-start", overflowY: "scroll"}}>

                    {this.state.messages.map((m, i) => m.isBot ? 
                        <div key={i} className="chatBubble botMsg">{m.msg}</div> :
                        <div key={i} className="chatBubble userMsg">{m.msg}</div>
                    )}
                    <div ref={(end) => { this.messagesEnd = end }} />

                </div>
                
                <div className="container row" style={{height: "auto"}}>

                    <textarea 
                        autoFocus
                        ref={(inp) => { this.inputField = inp }}
                        value={this.state.msg}
                        placeholder="Send a message"
                        onChange={this.handleInputChange}
                        onKeyDown={k => {if (k.key === "Enter") {this.sendMessageAndGetReply()}}}
                        style={{width: "75%", height: "42px", borderRadius: "5px", resize: "none"}}
                    />

                    <button 
                        className="blue-button"
                        onClick={async () => {this.sendMessageAndGetReply()}}
                        style={{width: "25%", marginLeft: "2px"}}
                    >Send</button>

                </div>

            </div>
        )
    }

}

export default SMSTestBox;
