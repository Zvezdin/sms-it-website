import React from "react"

import "../styles/default.css"


/**
 * Displays an error alert over the current page.
 * Consists of a large popup with error information and a dismiss button.
 * 
 * @param {string} name - the name of the error to display in the popup.
 * @param {string} message - the error message to display in the popup.
 * @param {function} _callback - reference to the function to call when 'Dismiss' is clicked.
 * 
 * @returns error popup to cover the screen.
 */
class ErrorAlert extends React.Component {

    render() {
        return (
            <div className="error-overlay container column">

                <p style={{fontSize: "1.3em", fontWeight: "600", color: "crimson", alignSelf: "center", margin: "2%"}}>Unexpected Error</p>

                <div className="horizontal-line" />

                <div className="container column">
                    <p data-testid="nameLabel" style={{fontWeight: "600", alignSelf: "flex-start", margin: "2%"}}>{this.props.name}</p>
                    <div data-testid="messageLabel" style={{overflowY: "scroll", height: "100%", width: "100%", margin: "5px"}}>

                        {this.props.message}

                        <br /><div className="horizontal-line" />

                        <p>If you would like to help us troubleshoot this, we have collected some diagnostics! Send us the following file at <i style={{fontWeight: "600"}}>team@sms-it.io</i></p>
                        <p style={{textAlign: "center"}}>Windows: <span className="filepath">*User*\\AppData\\Roaming\\sms-it\\logs\\renderer.log</span></p>
                        <p style={{textAlign: "center"}}>macOS: <span className="filepath">~/Library/Logs/sms-it/renderer.log</span></p>
                        <p style={{textAlign: "center"}}>Linux: <span className="filepath">~/.config/sms-it/logs/renderer.log</span></p>

                    </div>
                </div>

                <button 
                    className="blue-button"
                    style={{height: "20%", width: "30%", margin: "1% 35%"}}
                    onClick={this.props._callback}
                >DISMISS</button>
                
            </div>
        )
    }

}

export default ErrorAlert;
