import React, { useEffect } from "react"
import ReactTooltip from 'react-tooltip';

import { notify } from "../helper_functions"

import "../styles/default.css"

import store from "../Store"


/**
 * Displays a button with a phone number and its linked bot. A free number will be in blue with green "Available" text, and an 
 * occupied number will be greyed out with grey "In Use: ..." text specifying the bot occupying that number. Occupied numbers have
 * a tooltip to say they are occupied.
 * 
 * @param {string} number - the phone number to display.
 * @param {function} handleAvailable - reference to the function to call when an available number is clicked.
 * @param {function} showError - a function to display an error on the parent page of this component, this is called when an error occurs.
 * 
 * @returns a button with a phone number and text for either the bot tied to it or "Available".
 */
function PhoneButton(props) {
    let occupied = store.isNumberOccupied(props.number);

    useEffect(() => {
        ReactTooltip.rebuild()
    })

    return (
        <>
            <button 
                key={props.number} 
                data-for={occupied ? "phone_tip" : null}
                data-tip={occupied ? "This number is occupied" : null}
                onClick={async () => {

                    try {
                        if (!occupied) {
                            props.handleAvailable(props.number)
                        } else {
                            notify(
                                "danger",
                                "Cannot deploy chatbot to this number!",
                                "To use this number, you must first unlink the chatbot tied to it",
                                7500
                            )
                        }
                    } catch (e) {
                        props.showError()
                    }

                }} 
                className={
                    occupied
                    ? "gradient-box-greyed"
                    : "gradient-box"
                }
            >

                <p data-testid="numberLabel" className="phone-list-number">
                    {props.number}
                </p>

                {
                occupied 
                ?
                    <p data-testid="usageLabel" className="phone-list-usage in-use">
                        {"In Use: " + store.findBotByNumber(props.number).name}
                    </p> 
                :
                    <p data-testid="usageLabel" className="phone-list-usage available">
                        Available
                    </p>
                }

            </button>
        </>
    )

}

export default PhoneButton;
