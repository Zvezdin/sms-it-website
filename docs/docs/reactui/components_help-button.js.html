import React from "react"

import web from "../images/web.png"
import "../styles/default.css"


/**
 * Displays a help button in the top left corner of the window.
 * Button has a web icon and help text, and opens a new window with the specified URL.
 * 
 * @param {string} url - the URL for the help website/documentation.
 * 
 * @returns help button in top left position.
 */
class HelpButton extends React.Component {

    render() {
        return ( 
            <button 
                className="help-button" 
                onClick={() => {
                    const newWindow = window.open(this.props.url, "Help", "noopener,noreferrer")
                    if (newWindow) newWindow.opener = null
                }}
            >
                <img draggable="false" src={web} alt="Docs" />
                <span>HELP</span>
            </button>
        )
    }

}

export default HelpButton;
