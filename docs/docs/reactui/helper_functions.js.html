import { store as notif } from 'react-notifications-component'


/**
 * Displays a notification in the bottom center of the screen with varying colours.
 * Uses 'react-notifications-component' to implement the notifications.
 * 
 * @param {string} type - Can be 'success' (green), 'danger' (red), 'info' (teal), 'default' (blue), 'warning' (yellow).
 * @param {string} title - The text to show in bold at the top of the notification.
 * @param {string} message - The more detailed text to show at the bottom of the notification.
 * @param {number} duration - Duration (in milliseconds) for the notification to be visible on screen.
 * 
 * @returns an id for the notification.
 */
export function notify(type, title, message, duration) {
    return notif.addNotification({
        title: title,
        message: message,
        type: type,
        container: "bottom-center",
        animationIn: ["animation__animated", "animate__fadeIn"],
        animationOut: ["animate__animated", "animate_fadeOut"],
        dismiss: {
            duration: duration
        }
    })
}


/**
 * Increments the msgIdx value (in the state of a page) every 750ms. The msgIdx value is responsible 
 * for cycling through the processing messages when the loading spinner is shown on screen.
 * 
 * @param {object} page - the page object on which to increment the msgIdx.
 */
export function cycleProcessingMessages(page) {
    page.timeout = setInterval(() => {
        let currentIdx = page.state.msgIdx
        page.setState({ msgIdx: currentIdx + 1 })
    }, 750)
}
