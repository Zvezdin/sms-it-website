import React from "react"

import StageBar from "../components/stage-bar"
import SMSTestBox from "../components/sms-testbox"
import HelpButton from "../components/help-button"

import arrow from "../images/left-chevron.png"
import "../styles/default.css"

import store from "../Store"


/**
 * Responsible for rendering the test page on screen, where the user can test the selected chatbot 
 * both in and out of the deployment stages. Consists of 'Phone Simulator: <bot name>' heading and 
 * the sms-testbox component.
 */
class Test extends React.Component {
    helpURL = "https://sms-it.io/project/for-users/06-test/"

    render() {
        return (
            <>
                <div className="column center-items" style={{height: "15%"}}>
                    {
                        this.props.location.state.deploy
                        ? <h2>Deploy {store.candidateBot.name}</h2>
                        : <h2>Phone Simulator: {store.candidateBot.name}</h2>
                    }
                </div>

                <div className="container column" style={{height: "85%", justifyContent: "flex-start"}}>

                    {this.props.location.state.deploy ? <StageBar current={0} /> : null}

                    <SMSTestBox />

                </div>

                <div className="icon-container" style={{left: 0}}>
                    <img draggable="false" src={arrow} alt="Back" className="arrow-icon" onClick={() => {this.props.history.push("/manage")}} />
                </div>

                {
                    this.props.location.state.deploy
                    ? (
                        <div className="icon-container" style={{right: 0}}>
                            <img draggable="false" src={arrow} alt="Next" className="arrow-icon rotate180" onClick={() => {this.props.history.push("/deploy")}} />
                        </div>
                    ) : null
                }

                <HelpButton url={this.helpURL} />
            </>
        )
    }

}

export default Test;
